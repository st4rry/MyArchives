# To enable remote desktop without password
reg add HKLM\SYSTEM\CurrentControlSet\Control\Lsa /v LimitBlankPasswordUse /t REG_DWORD /D 0

# To disable remote desktop without password
reg delete HKLM\SYSTEM\CurrentControlSet\Control\Lsa /v
